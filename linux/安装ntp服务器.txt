#!/bin/bash
for HOST in `cat hosts`
do
echo "install ntp"
ssh -t root@$HOST "sudo yum -y install ntp"
ssh -t root@$HOST "sudo systemctl enable ntpd"
ssh -t root@$HOST "sudo systemctl start ntpd"
ssh -t root@$HOST "sudo ntpdate -u cn.pool.ntp.org"
echo "$HOST ntp installed"
done
